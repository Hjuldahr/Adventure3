package com.example.adventure.action;

import com.example.adventure.creature.Creature;

import java.util.List;

import com.example.adventure.action.DamageType.DamageTypes;
import com.example.adventure.creature.AbilityScores.AbilityCategories;
import com.example.adventure.utility.Dice;
import com.example.adventure.utility.DicePool;
import com.example.adventure.utility.Success;
import com.example.adventure.utility.Success.LuckTypes;
import com.example.adventure.utility.Success.SuccessTypes;

public abstract class Spell {
    public static enum ActivationType {
        ACTION,
        BONUS_ACTION,
        FREE_ACTION, // consumes no action
        REACTION // requires an external reaction trigger
    };
    public static enum AreaType {
        SELF,
        SINGLE_TARGET, 
        MULTI_TARGET, //manual selection
        AOE //all targets
    }
    public static enum Category {
        ATTACK, 
        HARM, // spell save instead of hit
        HEALING, 
        SUMMONING, 
        WARDING, 
        BLESSING,
        CURSING
    }

    private Category categoryType;

    private String name;

    private int spellLevel;

    private ActivationType activationType;

    private AreaType areaType;
    private int multiTargetLimit;

    private int duration; // measured in turns. 0 if instant

    private AbilityCategories saveType;

    // effect is always applied on hit or fail
    private Effect effect;

    private boolean halfDamageOnSave; // halve harm damage on passed saves
    private boolean effectOnSave; // only apply affect on failed saves vs everytime

    private DicePool dicePool;
    private DamageTypes damageType;

    private boolean requiresConcentration;

    private boolean automaticallHits;

    public Spell() {

    }

    public String getName() {
        return name;
    }

    // Assumes enemies vs allies will be correctly picked for heals vs attacks
    public void use(Creature caster, List<Creature> targets) {
        if (requiresConcentration) {
            // automatically breaks previous spell concentration
            caster.setConcentratedSpell(this);
        }

        switch (areaType) {
            case SELF: // enforce limit
                applyPerTarget(caster, caster);
                break;
            case SINGLE_TARGET: // enforce limit
                applyPerTarget(caster, targets.getFirst());
                break;
            case MULTI_TARGET: // enforce limit
                targets.subList(0, multiTargetLimit).forEach(target -> applyPerTarget(caster, target));
                break;
            case AOE: // everything
                targets.forEach(target -> applyPerTarget(caster, target));
                break;
        }
    }

    // summons are unique as it requires instantiating transient (per encounter) entites
    private void applyPerTarget(Creature caster, Creature target) {
        switch (categoryType) {
            case ATTACK: // spell attack check
                attackTarget(caster, target);
                break;
            case HARM: // spell save
                harmTarget(caster, target);
                break;
            case HEALING:
                healTarget(caster, target);
                break;
            case WARDING:
                wardTarget(caster, target);
                break;
            case BLESSING:
                blessTarget(caster, target);
                break;
            case CURSING:
                curseTarget(caster, target);
                break;
            default:
                break;
        }
    }

    // Distinct from AoE damage
    private int rollAttackDamage(LuckTypes luckType) {
        if (luckType == LuckTypes.TRIUMPH) {
            return dicePool.rollMax();
        } else if (luckType == LuckTypes.FUMBLE) {
            return dicePool.rollMin();
        } 
        return dicePool.rollAll();
    }

    // For attack spells (firebolt)
    private boolean attackTarget(Creature caster, Creature target) {
        int raw = Dice.d20();
        int result = raw + caster.getSpellCastingModifier() + caster.getProfiencyBonus();
        
        int armourClass = target.getArmourClass();
        
        SuccessTypes successType = Success.evaluateSuccess(result, armourClass);
        LuckTypes luckType = Success.evaluateLuck(raw);

        if (successType == SuccessTypes.FAILURE || successType == SuccessTypes.CRIT_FAILURE) {
            return false;
        } 
        
        int damageRoll = rollAttackDamage(luckType); 
        if (successType == SuccessTypes.CRIT_SUCCESS) {
            damageRoll += rollAttackDamage(luckType);
        }

        target.applyDamage(damageRoll, damageType); // internally calculate resistance, immunity, vulnerability

        if (effect != null) {
            target.applyEffect(effect);
        }

        return true;
    }

    // For auto hit spells (magic missile)
    private void harmTarget(Creature caster, Creature target) {
        int damage = dicePool.rollAll();
        target.applyDamage(damage, damageType); // internally calculate resistance, immunity, vulnerability

        // for the effect not the attack
        int raw = Dice.d20();
        int result = raw + target.getAbilityModifier(saveType) + target.getSaveModifier(saveType);
        int difficultyClass = caster.getSpellSaveDifficulty();

        if (effectOnSave && effect != null && !Success.simpleEvaluateSuccess(result, difficultyClass)) {
            target.applyEffect(effect);
        }
    }

    // For AoEs (fireball)
    private void saturateTarget(Creature caster, Creature target) {
        float damageModifier = 1f;
        
        // for the attack
        int difficultyClass = caster.getSpellSaveDifficulty();
        int raw = Dice.d20();
        int result = raw + target.getAbilityModifier(saveType) + target.getSaveModifier(saveType);
        
        SuccessTypes successType = Success.evaluateSuccess(result, difficultyClass);
        LuckTypes luckType = Success.evaluateLuck(raw);
        
        // crit pass + trumph = x0 dmg
        // crit fail + fumble = x2 dmg
        damageModifier = switch (successType) {
            case CRIT_SUCCESS -> 0.5f; // x1/2 dmg
            case SUCCESS -> 0.5f; // x1/2 dmg
            case FAILURE -> 1f;
            case CRIT_FAILURE -> 1.5f; 
        } + switch (luckType) {
            case TRIUMPH -> -0.5f;
            case NONE -> 0f;
            case FUMBLE -> 0.5f;
        };
        damageModifier = Math.clamp(damageModifier, 0f, 2f);

        int damage = (int) Math.floor(dicePool.rollAll() * damageModifier);
        target.applyDamage(damage, damageType); // internally calculate resistance, immunity, vulnerability

        if (effectOnSave && effect != null && (successType == SuccessTypes.FAILURE || successType == SuccessTypes.CRIT_FAILURE)) {
            target.applyEffect(effect);
        }
    }

    private void healTarget(Creature caster, Creature target) {
        int raw = Dice.d20();
        
        if (raw == 1) return;
        
        int healRoll = dicePool.rollAll() + caster.getSpellCastingModifier();

        if (raw == 20) {
            healRoll += dicePool.rollMax();
        }
        
        target.applyHeal(healRoll);

        if (effect != null) {
            target.applyEffect(effect);
        }
    }

    private void wardTarget(Creature caster, Creature target) {
        if (effect != null) {
            target.applyEffect(effect);
        }
    }

    private void blessTarget(Creature caster, Creature target) {
        if (effect != null) {
            target.applyEffect(effect);
        }
    }

    private void curseTarget(Creature caster, Creature target) {
        int difficultyClass = caster.getSpellSaveDifficulty();
        if (!target.simpleSaveCheck(difficultyClass, saveType)) return;
        
        if (effect != null) {
            target.applyEffect(effect);
        }
    }

    public void concentrationBroken() {
        if (effect != null && requiresConcentration) {
            effect.removeAllEntities();
        }
    }

    public boolean getRequiresConcentration() {
        return requiresConcentration;
    }

    public Effect getEffect() {
        return effect;
    }
}
